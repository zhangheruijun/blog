3、legend: 图例组件 #

图例组件展现了不同系列的标记(symbol)，颜色和名字。可以通过点击图例控制哪些系列不显示。比如柱状图有多组的时候

ts
{
  //plain：普通图例，默认就是; scroll: 可滚动翻页的图例。当图例数量较多时可以使用。
  type: 'plain' | 'scroll'

  // 布局朝向,vertical:垂直 horizontal:水平
  orient: 'vertical' | 'horizontal',

  // 图例组件离容器左侧的距离,还有right、top、bottom; 默认都是'auto'
  left: string | number

  //图标icon样式,也可以设置图片
  icon:'circle'|'rect'|'roundRect'|'triangle'|'diamond'|'pin'|'arrow'|'none'

  // 图形设置,设置图形描边、边框等
  itemStyle:{
    color:'transparent', // 设置背景颜色为透明
    borderColor:'red', // 再设置边框颜色为红色,
    ...
  }

  // 设置图形旁边的文字
  textStyle:{
    color:'blue',
    fontSize: number
  },

  // 图形线的设置,用于诸如折线图图例横线的样式设置
  lineStyle:{...}

}

4、xAxis: X轴 #

ts
{
  /* 自动计算的坐标轴最小间隔大小,以设置成1保证坐标轴分割刻度显示成整数
  只在数值轴或时间轴中（type: 'value' 或 'time'）有效。
  如果设置为0, 该轴是连续的，中间不会断
  */
  minInterval: number

  // 坐标轴 轴线 相关设置
  axisLine:{
    lineStyle:{
      color:'red' // 设置坐标轴为红色
    }
  }

  // 坐标轴 刻度 相关设置
  axisTick:{
    lineStyle:{
      color:'yellow' // 设置刻度为黄色
    }
  }
  
  // 坐标轴在 grid 区域中的分隔线。
  splitLine:{
    show: false // 关掉x轴的网格线，Y轴同样也是这个api
  }

  // 坐标轴 标签(文字) 相关设置
  axisLabel:{
    // 标签(文字)旋转,-90 度到 90 度
    rotate: number

    // 比如文字多了要换行显示,通过计算得到;比如 params = '2022-09-23上学期分数'
    formatter (params:string) {
      let newParamsName = ''
      const strLen = params.length // 总字数
      const PerRowNum = 9 // 一行显示几个字
      const rowNumber = Math.ceil(strLen / PerRowNum) // 显示几行
      if (strLen > PerRowNum) {
        for (let p = 0; p < rowNumber; p++) {
          let tempStr = ''
          const start = p * PerRowNum
          const end = start + PerRowNum
          if (p === rowNumber - 1) {
            tempStr = params.substring(start, strLen)
          } else {
            tempStr = params.substring(start, end) + '\n'
          }
          newParamsName += tempStr
        }
      } else {
        newParamsName = params
      }
      return newParamsName
    }
  }
}

ts
{
  // line: 折线图 bar: 柱状图 pie: 饼图
  type: string

  // 图像样式设置,各种图形不通,会有不同,参考官网
  itemStyle: {
    // type为bar时, 设置柱子背景渐变色
    color: {
      type: 'linear',
      x: 0,
      y: 0,
      x2: 0,
      y2: 1,
      colorStops: [
        // 柱子渐变颜色
        {
          offset: 0,
          color: custColors[index][0], // 0% 处的颜色
        },
        {
          offset: 0.5,
          color: custColors[index][1], // 100% 处的颜色
        },
        {
          offset: 1,
          color: custColors[index][2], // 100% 处的颜色
        },
      ],
      global: false, // 缺省为 false
    },
    borderRadius: [20, 20, 0, 0], // 柱子圆角
  }


  // 轴滑动
  dataZoom:[
    {
    xAxisIndex: 0, //这里是从X轴的0刻度开始
    show: false, //是否显示滑动条，不影响使用
    type: 'inside', // 这个 dataZoom 组件是 slider 型 dataZoom 组件
    startValue: 0, // 从头开始。
    endValue: 4, // 到哪里结束    startValue <= 展示的个数 < endValue
    },
    {
      startValue: 0, // 从头开始。
      endValue: 4, // 一次性展示几个。
      height: 15, //组件高度
      // left: '5%', //左边的距离
      // right: '5%',//右边的距离
      bottom: '6%', //下边的距离

      // showDetail: false,//即拖拽时候是否显示详细数值信息 默认true
      // realtime: true, //是否实时更新
      // filterMode: 'filter',

// 设置自动滚动,endValue: 次性展示几个
const setScroll = (endValue: number) => {
  timer = setInterval(function () {
    if (isMouseMove.value) return;
    // 每次向后滚动一个，最后一个从头开始
    if (option.dataZoom[0].endValue === xAxisList.length) {
      option.dataZoom[0].startValue = 0; // 数据窗口范围的起始数值
      option.dataZoom[0].endValue = endValue; // 数据窗口范围的结束数值
    } else {
      option.dataZoom[0].startValue = option.dataZoom[0].startValue + 1; // 数据窗口范围的起始数值
      option.dataZoom[0].endValue = option.dataZoom[0].endValue + 1; // 数据窗口范围的结束数值
    }
    myChart.setOption(option);
  }, 3000);
};
