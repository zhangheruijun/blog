// ----------------------------------精确匹配------------------------------
// 普通匹配
console.log(/zhang/.test('bbzhangevv')); // true 必有有连着的字符'zhang'才行,代表一位

// 元字符：是有特殊语意的，如果想正常匹配元字符的字符串得用转译，转译用 \
// 元字符有 . [] [^] ? * + {min,max} ^ $ () | / 等
// 中文正则表达 [\u4E00-\u9FA5]
console.log(/\[/.test('zhang[kk')); // true


// ---------------------------------修饰符----------------------------
/**
 *  i  ignore - 不区分大小写, 将匹配设置为不区分大小写，搜索时不区分大小写: A 和 a 没有区别。
 *  g  global - 全局匹配，查找所有的匹配项。
 *  m  multi line - 多行匹配，使边界字符 ^ 和 $ 匹配每一行的开头和结尾，记住是多行，而不是整个字符串的开头和结尾。
 *  s  特殊字符圆点 . 中包含换行符 \n, 意思是: 默认情况下的圆点 . 是匹配除换行符 \n 之外的任何字符，加上 s 修饰符之后, . 中包含换行符 \n。
 */




// -----------------------------字符组(匹配范围)------------------------------
/**
 * 字符组：[] 在固定的范围匹配一个就行; []代表一位
 * [^] 是取反的意思
 * /[zhang]/ 只要存在'zhang'任何一个字符就行
 * /[a-z]/ 只要存在a到z任何一个字符就行,包含 a 和 z
 * /[A-Z]/ 只要存在A到Z任何一个字符就行,包含 A 和 Z
 * /[0-9]/ 只要存在0到9任何一个字符就行,包含 0 和 9
 * /[0-9A-Z]/ 只要存在0到9或者A-Z任何一个字符就行,包含 0、9、A、Z
 * 字符组中的元字符不需要转译：/[./\?*\[\]]/ 字符组里面就不需要转译,如果单独匹配[]需要转译,如果不转义[]代表一个字符
 * 整个匹配 \[\],多个匹配尽量在字符组外面   单个匹配 \]\[ ☆☆☆☆☆☆
 * ...
 */

/**
 * 简写
 * \d 等价于 [0-9]
 * \D 等价于 [^0-9]
 * \w 等价于 [A-Za-z0-9_] 字母、数字、下划线 
 * \W 等价于 [^A-Za-z0-9_] 非字母、数字、下划线 
 * \s 等价于 [\f\n\r\t\v] 匹配任何空白字符 console.log('\r') 打印结果为空
 * \S 等价于 [^\f\n\r\t\v] 匹配任何非空白字符 console.log('\r') 打印结果为空
 */
//  \n 换行符  \r 回车符  \t 制表符  \f 换页符
// 字符串'bbzhangevv'只要存在'zhang'任何一个字符就行
console.log(/[zhang]/.test('bbzhangevv')); // true

// 字符串'11554'的4在范围内
console.log(/[2-4]/.test('11554')); // true

// 字符串'chdf'的'd'与'f'在'c'与'h'的反面,所以成立
console.log(/[^ch]/.test('chdf')); // true

// 分开配 ] [ 要分开写, 整个匹配则: \[\]
console.log(/[\][]/.test('[')); // true






// ----------------------------------限定符(量词)-------------------------------
/** 一般限制前一位，
 * n{4} 代表匹配紧挨着的有4个n，比如 15428nnnnb878565
 * [2-5]{3} 代表匹配紧挨着的有3个2-5的字符，比如 zhas354dsfg
 * 
 * n{4,} 代表匹配紧挨着至少有4个n，n的个数要 n >= 4
 * n{4,7} 代表匹配紧挨着至少有4个到7个之间的n，n的个数要  4 <= n <= 7
*/

/**
 * 简写
 * ? 等同于 {0,1}
 * * 等同于 {0,}
 * + 等同于 {1,}
 * . 匹配除换行符（\n、\r）之外的任何单个字符。要匹配包括 '\n' 在内的任何字符，请使用像"(.|\n)"的模式,\r表示回车。
 * . 特殊字符在中括号表达式时 如 [.] 只会匹配 .字符，等价于 \.，而非匹配除换行符 \n 外的所有字符。默认情况下的圆点 . 是 匹配除换行符 \n 之外的任何字符，加上 s 之后, . 中包含换行符 \n。
*/

// 等同于/[a-d][a-d]/，要紧挨着两位是 a-d 之间的字符
console.log(/[a-d]{2}/.test('15428cb878565'));

// 正则默认是贪婪模式，能多个匹配就多匹配
// 匹配到 'bbbbb'; 按理说匹配到2个就可以了
// 是往最大值匹配
console.log(/b{2,}/.exec('15428bbbbb878565')); // ['bbbbb', index: 5, input: '15428bbbbb878565', groups: undefined]


// 非贪婪模式，匹配最小的的那个
// 后面加上 ?  就是非贪婪模式
// 非贪婪模式必须在在限定符后面加
console.log(/a??/.exec('aaaaa')); // ['', index: 0, input: 'aaaaa', groups: undefined]
console.log(/a*?/.exec('aaaaa')); // ['', index: 0, input: 'aaaaa', groups: undefined]
console.log(/a+?/.exec('aaaaa')); // ['a', index: 0, input: 'aaaaa', groups: undefined]
console.log(/a{1,3}/.exec('aaaaa')); // ['aaa', index: 0, input: 'aaaaa', groups: undefined]
console.log(/a{1,3}?/.exec('aaaaa')); // ['a', index: 0, input: 'aaaaa', groups: undefined]
// 限定符也可以放在括号后,下面?前面的括号代表一个整体
console.log(/(?:\d{3})?/.test('123')); // true ☆☆☆☆☆☆
console.log(/(?:\d{3})?/.test('')); // true ☆☆☆☆☆☆




// ----------------------------------()分组、捕获-------------------------------

// 分组：分组的同时又能捕获子表达式；一箭双雕, 分组后面加上量词代表重复该括号内几次
// 捕获：括号包的谁,谁就子表达式,谁就是子表达式
// 分组的同时又能捕获，如果捕获相同,如: (a){2} 等同于 (a)(a) 取最后一位,无论取哪一位都一样，☆☆☆☆☆☆
// 如果不相同 ([2-8]){2}; 等同于([2-8])([2-8]);这种不同，只能算最后一位 ☆☆☆☆☆☆
console.log(/\d{4}-\d{1,2}-\d{1,2}/.exec('2022-02-05')); // ['2022-02-05', index: 0, input: '2022-02-05', groups: undefined]

console.log(/(\d{4})-(\d{1,2})-(\d{1,2})/.exec('2022-02-05')); // ['2022-02-05', '2022', '02', '05', index: 0, input: '2022-02-05', groups: undefined]

console.log(/(a){2}-(b){2}-(c){2}/.exec('aa-bb-cc')); // ['aa-bb-cc', 'a', 'b', 'c', index: 0, input: 'aa-bb-cc', groups: undefined]

console.log(/(a{2})-(b{2})-(c{2})/.exec('aa-bb-cc')); // ['aa-bb-cc', 'aa', 'bb', 'cc', index: 0, input: 'aa-bb-cc', groups: undefined]

console.log(/ab{2}/.exec('abb')); // ['abb', index: 0, input: 'abb', groups: undefined]

console.log(/(ab{2})/.exec('abb')); // ['abb', 'abb', index: 0, input: 'abb', groups: undefined]

console.log(/(ab){2}/.exec('abab')); // ['abab', 'ab', index: 0, input: 'abab', groups: undefined]




// --------------------------------------反向应用、非捕获-----------------------------------------
/**
 *  反向应用: 下一个匹配需要应用上一个
*/


console.log(/([a-z])/.exec('df')); // ['d', 'd', index: 0, input: 'df', groups: undefined]

// 其中 \1 代表反向应用 ([a-z]) 子表达式中第一个，
console.log(/([a-z])\1/.exec('ddfg'));


/**
 * 非捕获: 就是取消子表达式; 语法为：在括号里面的前面加 ?:
*/

// 子表达中第一项为 abc
console.log(/(abc){2}/.exec('abcabc12'));  // ['abcabc', 'abc', index: 0, input: 'abcabc12', groups: undefined]

// 子表达中第一项是没有的
console.log(/(?:abc){2}/.exec('abcabc12'));  // ['abcabc', index: 0, input: 'abcabc12', groups: undefined]



// ------------------------------------分支匹配----------------------------------------
/**
 * 语法为：|   类似于 || 
 * 匹配规则: 从左到右匹配，一旦匹配到就不在回头
*/


console.log(/ab|cd|45/.exec('ab'));  // ['ab', index: 0, input: 'ab', groups: undefined]
console.log(/ab|cd|45/.exec('cd'));  // ['cd', index: 0, input: 'cd', groups: undefined]
console.log(/ab|cd|45/.exec('45'));  // ['45', index: 0, input: '45', groups: undefined]

// 从左到右匹配，只要匹配到就停止
console.log(/ab|cd|45/.exec('abcd45'));  // ['ab', index: 0, input: 'abcd45', groups: undefined]

// 匹配0-4 或者 7-9之间的任意一个数
console.log(/[0-4]|[7-9]/.exec('g8')); // ['8', index: 1, input: 'g7', groups: undefined]

// 括号匹配,重点: 有括号的地方就有子表达式,可以用 ?: 取消捕获
// 相当于 /abd|45/ 或者 /acd|45/
console.log(/a(b|c)d|45/.exec('abd')); // ['abd', 'b', index: 0, input: 'abd', groups: undefined]


// 匹配 0~255 之间的整数,要分段匹配
/**
 * 一位数： 0~9   [0-9]
 * 两位数： 10~99  [1-9][0-9]
 * 三位数： 100~255 
 *          100~199 1[0-9][0-9]
 *          200~249 2[0-4][0-9]
 *          250~255 25[0-5]
*/

// 匹配到 2，因为遇到第一个符合就停止了，
console.log(/[0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5]/.exec('241')); // ['2', index: 0, input: '255', groups: undefined]

// 所以得倒着写
console.log(/25[0-5]|2[0-4][0-9]|1[0-9][0-9]|[1-9][0-9]|[0-9]/.exec('241')); // ['254', index: 0, input: '254', groups: undefined] 

// 或者用开始结束符号
console.log(/^[0-9]$|^[1-9][0-9]$|^1[0-9][0-9]$|^2[0-4][0-9]$|^25[0-5]$/.exec('254')); // ['254', index: 0, input: '254', groups: undefined] 




// ------------------------------------断言(边界)-------------------------------------
// 匹配一个单词边界，也就是指单词和空格间的位置。例如， 'er\b' 可以匹配"never" 中的 'er'，但不能匹配 "verb" 中的 'er'。
// 详细解释：https://www.cnblogs.com/litmmp/p/4925374.html  ☆☆☆☆☆☆
// 更精确的说法，\b 匹配这样的位置：它的前一个字符和后一个字符不全是(一个是,一个不是或不存在) \w。  ☆☆☆☆☆☆

//相当于匹配ab, 但是ab前面得是非\w
console.log(/\bab/.test('c abc12')); // true
console.log(/\bab/.test('cabc12')); // false
console.log(/\bab/.test('abc12')); // true
console.log(/a\b.\bnice/.test('a nice')); // true
// 正则错误,因为\b前后都为\w类型
console.log(/a\bnice/.test('a nice')); // false


// ab前后得有空格或者没有字符，
console.log(/\bab\b/.test('c ab c12')); // true
console.log(/\bab\b/.test('cab c12')); // false

// 开始符 ^  与结束符 $, 可以用来限定个数
console.log(/^zh/.test('zhdf')); // true
console.log(/^zh/.test('qzhdf')); // false

console.log(/fg$/.test('sdfsdfg')); // true
console.log(/fg$/.test('sdfsdfgp')); // false


// --------------------------------环视--------------------------
/**
 * 在()里面使用,环视()里没有子表达式
 * ?=、?<=、?!、?<!
*/

// exp1(?=exp2)：查找 exp2 前面的 exp1, 相当于查找exp1，但是exp1后面必须是exp2
console.log(/ruonb(?=[0-9])/.test('sdruonb2fgfruonb6g')); // true
console.log('sdruonb2fgfruonb3g'.replaceAll(/ruonb(?=[0-9])/g, '-')); // sd-2fgf-3g


// exp1(?!exp2)：查找后面不是 exp2 的 exp1。相当于查找exp1，但是exp1后面不能是exp2
console.log(/ruonb(?![0-9])/.test('sdruonb2fgfruonbPPg')); // true
console.log('sdruonb2fgfruonbPPg'.replaceAll(/ruonb(?![0-9])/g, '-')); // sdruonb2fgf-PPg


// (?<=exp2)exp1：查找 exp2 后面的 exp1。相当于查找exp1，但是exp1前面必须是exp2
console.log(/(?<=[0-9])ruonb/.test('sd2ruonbfgf34ruonb6g')); // true
console.log('sd2ruonbfgf34ruonb6g'.replaceAll(/(?<=[0-9])ruonb/g, '-')); // sd2-fgf34-6g


// (?<!exp2)exp1：查找前面不是 exp2 的 exp1。相当于查找exp1，但是exp1前面不能是exp2
console.log(/(?<![0-9])ruonb/.test('sd45ruonb2fgfruonbPPg')); // true
console.log('sd45ruonb2fgfruonbPPg'.replaceAll(/(?<![0-9])ruonb/g, '-')); // sd45ruonb2fgf-PPg


// ---------------------------------修饰符 m ----------------------------------
/**
 * hhzhang\n大萨达 是两行如下
 * hhzhang
 * 大萨达
 * 正则匹配的是所有，换行也算在内部，如果换行单独算用修饰符 m
*/
console.log(/zhang$/.test('hhzhang\n大萨达')); // false
console.log(/萨达$/m.test('hhzhang\n大萨达\n健健康康')); // true
console.log(/^大萨/m.test('hhzhang\n大萨达\n健健康康')); // true


// ---------------------------------修饰符 s ----------------------------------
/**
 * 默认情况下的圆点 . 是匹配除换行符 \n 之外的任何字符，加上 s 修饰符之后, . 中包含换行符 \n。  ☆☆☆☆☆☆
 */
console.log(/大萨达./s.test('hhzhang\n大萨达\n健健康康')); // true    ☆☆☆☆☆☆



/**
 * 断言高级特性
 * 零宽断言：断言条件本身不会消耗字符
 * 多边界断言: 一个正则可同时存在多个边界断言。根据其所处位置来决定断言的边界。
  条件组合: 多个子断言可以进行布尔运算如：&& || ()，
    h(?=条件1)(?=条件2) 且运算，指必须同时满足多个条件
    h(?=(?=条件1)(?=条件2)) 且运算，条件 1 与条件 2 组成一个新条件，并且两个条件都必须满足
    h(?=(?=条件1)|(?!条件2)) 或运算，满足条件 1 或者 不满足 条件 2
    h(?=表达式1|(?=条件1)(?!条件2)) 混合运算，表示满足表达式 1 或者同时满足条件 1、2
    
    详情： https://learnku.com/articles/70841
 * */

// 密码校验1：最少6位，包括至少1个大写字母，1个小写字母，1个数字，1个特殊字符
// 意思是长度大于6，只有四种类型，那么就成立
/**
 * @ 代表任意类型
 * 解析: 等同于 .*(?=.{6,}) 且 (?=.*\d) ...
 * 1. .(?=.{6,}) 代表：@ {@@@@@@}
 * 2. (?=.*\d)  代表@ {@@@@@@} 同时还要满足1的要求
 */
console.log(/^.*(?=.{6,})(?=.*\d)(?=.*[A-Z])(?=.*[a-z])(?=.*[!@#$%^&*?]).*$/.test('aA1^45dw111')); // true  ☆☆☆☆☆☆

// 这种写法也可以，只是用exce匹配不全
console.log(/(?=.{6,})(?=.*\d)(?=.*[A-Z])(?=.*[a-z])(?=.*[!@#$%^&*?])/.test('aA1^45dw111')); //true 

// 如果要过滤空格，要结合下面的用
console.log(/\s/.test(' abcxy zdef ')); // true   ☆☆☆☆☆☆

// 密码校验2：最少6位，里面只能是大写字母，小写字母，数字，特殊字符
// 意思是长度大于6，类型都是这四种，那么就成立
console.log(/^[a-zA-Z0-9!@#$%^&*?]{6,}$/.test('aA1^45dw111'));
